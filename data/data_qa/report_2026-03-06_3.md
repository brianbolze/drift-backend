## Queries Run

```sql
-- Referential integrity (all 0)
SELECT 'cartridge->bullet', COUNT(*) FROM cartridge c LEFT JOIN bullet b ON c.bullet_id = b.id WHERE c.bullet_id IS NOT NULL AND b.id IS NULL
-- (plus 4 more FK checks, all 0)

-- Weight mismatches (67 rows)
SELECT c.id, c.name, c.bullet_weight_grains, b.weight_grains, ABS(c.bullet_weight_grains - b.weight_grains) AS diff
FROM cartridge c JOIN bullet b ON c.bullet_id = b.id
WHERE ABS(c.bullet_weight_grains - b.weight_grains) > 1.0;

-- Diameter mismatches (0 rows)
SELECT c.id, c.name, b.bullet_diameter_inches, cal.bullet_diameter_inches, cal.name
FROM cartridge c JOIN bullet b ON c.bullet_id = b.id JOIN caliber cal ON c.caliber_id = cal.id
WHERE ABS(b.bullet_diameter_inches - cal.bullet_diameter_inches) > 0.002;

-- BC error thresholds (2 rows — both legitimate)
SELECT b.id, b.name, m.name, b.bullet_diameter_inches, b.bc_g1_published, b.bc_g7_published
FROM bullet b JOIN manufacturer m ON b.manufacturer_id = m.id
WHERE (b.bc_g1_published IS NOT NULL AND (b.bc_g1_published < 0.05 OR b.bc_g1_published > 1.2))
   OR (b.bc_g7_published IS NOT NULL AND (b.bc_g7_published < 0.02 OR b.bc_g7_published > 0.6));

-- Zero MV (7 rows — all Hornady International)
SELECT c.id, c.name, c.muzzle_velocity_fps, m.name
FROM cartridge c JOIN manufacturer m ON c.manufacturer_id = m.id
WHERE c.muzzle_velocity_fps <= 0 OR c.muzzle_velocity_fps > 5000;

-- Missing BCs excl CE/Nosler (6 rows, was 23)
SELECT b.id, b.name, m.name, b.bullet_diameter_inches
FROM bullet b JOIN manufacturer m ON b.manufacturer_id = m.id
WHERE b.bc_g1_published IS NULL AND b.bc_g7_published IS NULL
  AND b.bc_g1_estimated IS NULL AND b.bc_g7_estimated IS NULL
  AND b.bullet_diameter_inches <= 0.375
  AND m.name NOT LIKE '%Cutting Edge%' AND m.name NOT LIKE '%Nosler%';
```
