## New Critical Issues

### C8: Sako Gamehead .308 150gr — Wrong BC
- **Bullet ID**: `6eabdcb4-290c-4cc5-94bc-6ebb4d708997`
- **DB value**: G1 = 0.322
- **Correct value** (sako.global): G1 = **0.304**
- **Root cause**: Curation patch used the 30-06 Springfield 150gr BC (0.322) instead of the .308 Win 150gr BC (0.304). Both appear in the same product table on Sako's website with identical weight/diameter.
- **Impact**: 5.9% BC overestimate → will produce optimistic drop predictions at range
- **Fix**: Curation patch to update `bc_g1_published` from 0.322 → 0.304

-- Referential integrity (all zero)
SELECT 'orphan_bullet_mfr', COUNT(*) FROM bullet b LEFT JOIN manufacturer m ON b.manufacturer_id = m.id WHERE m.id IS NULL
UNION ALL SELECT 'orphan_cart_bullet', COUNT(*) FROM cartridge c LEFT JOIN bullet b ON c.bullet_id = b.id WHERE c.bullet_id IS NOT NULL AND b.id IS NULL
UNION ALL SELECT 'orphan_cart_caliber', COUNT(*) FROM cartridge c LEFT JOIN caliber cal ON c.caliber_id = cal.id WHERE cal.id IS NULL
UNION ALL SELECT 'orphan_cart_mfr', COUNT(*) FROM cartridge c LEFT JOIN manufacturer m ON c.manufacturer_id = m.id WHERE m.id IS NULL
UNION ALL SELECT 'orphan_bcs_bullet', COUNT(*) FROM bullet_bc_source bcs LEFT JOIN bullet b ON bcs.bullet_id = b.id WHERE b.id IS NULL
UNION ALL SELECT 'orphan_alias', COUNT(*) FROM entity_alias ea WHERE (ea.entity_type='manufacturer' AND ea.entity_id NOT IN (SELECT id FROM manufacturer)) OR (ea.entity_type='caliber' AND ea.entity_id NOT IN (SELECT id FROM caliber)) OR (ea.entity_type='bullet' AND ea.entity_id NOT IN (SELECT id FROM bullet));

-- Weight mismatches (8 found — known C1)
SELECT c.id, c.name, c.bullet_weight_grains, b.weight_grains, ABS(c.bullet_weight_grains - b.weight_grains)
FROM cartridge c JOIN bullet b ON c.bullet_id = b.id
WHERE ABS(c.bullet_weight_grains - b.weight_grains) > 1.0;

-- Diameter mismatches (0 found)
SELECT c.id, c.name, b.bullet_diameter_inches, cal.bullet_diameter_inches
FROM cartridge c JOIN bullet b ON c.bullet_id = b.id JOIN caliber cal ON c.caliber_id = cal.id
WHERE ABS(b.bullet_diameter_inches - cal.bullet_diameter_inches) > 0.002;

-- Zero/implausible MV (14 found — known W1)
SELECT c.id, c.name, c.muzzle_velocity_fps FROM cartridge c WHERE c.muzzle_velocity_fps <= 0 OR c.muzzle_velocity_fps > 5000;

-- Missing BC rifle bullets excl CE/Nosler (33 found — known W2)
SELECT m.name, COUNT(*) FROM bullet b JOIN manufacturer m ON b.manufacturer_id = m.id
WHERE b.bc_g1_published IS NULL AND b.bc_g7_published IS NULL AND b.bc_g1_estimated IS NULL AND b.bc_g7_estimated IS NULL
  AND b.bullet_diameter_inches <= 0.375 AND m.name NOT IN ('Cutting Edge Bullets','Nosler')
GROUP BY m.name;
