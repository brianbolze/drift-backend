The source URL (`sierrabullets.com/6-5mm-107-gr-tipped-matchking-tmk`) confirms this is a 6.5mm bullet. A separate correct 6mm (.243) record exists at ID `44644b15` with G1=0.574, G7=0.292. **Fix**: Rename to "6.5MM 107 GR Tipped MatchKing (TMK)" via curation patch.

```sql
-- Referential integrity (all 0)
SELECT 'bullets_bad_mfr', COUNT(*) FROM bullet b LEFT JOIN manufacturer m ON b.manufacturer_id = m.id WHERE m.id IS NULL
UNION ALL SELECT 'carts_bad_bullet', COUNT(*) FROM cartridge c LEFT JOIN bullet b ON c.bullet_id = b.id WHERE b.id IS NULL
UNION ALL SELECT 'carts_bad_caliber', COUNT(*) FROM cartridge c LEFT JOIN caliber cal ON c.caliber_id = cal.id WHERE cal.id IS NULL
UNION ALL SELECT 'carts_bad_mfr', COUNT(*) FROM cartridge c LEFT JOIN manufacturer m ON c.manufacturer_id = m.id WHERE m.id IS NULL
UNION ALL SELECT 'bcs_bad_bullet', COUNT(*) FROM bullet_bc_source bcs LEFT JOIN bullet b ON bcs.bullet_id = b.id WHERE b.id IS NULL;

-- Weight mismatches (7 found)
SELECT c.id, c.name, c.bullet_weight_grains, b.weight_grains, ABS(c.bullet_weight_grains - b.weight_grains)
FROM cartridge c JOIN bullet b ON c.bullet_id = b.id
WHERE ABS(c.bullet_weight_grains - b.weight_grains) > 1.0;

-- Diameter mismatches (0 found)
SELECT c.id, c.name, b.bullet_diameter_inches, cal.bullet_diameter_inches
FROM cartridge c JOIN bullet b ON c.bullet_id = b.id JOIN caliber cal ON c.caliber_id = cal.id
WHERE ABS(b.bullet_diameter_inches - cal.bullet_diameter_inches) > 0.002;

-- Zero/implausible MV (14 zero + 2 high)
SELECT c.id, c.name, c.muzzle_velocity_fps FROM cartridge c
WHERE c.muzzle_velocity_fps <= 0 OR c.muzzle_velocity_fps > 4200;

-- BC outliers (all within context-appropriate ranges)
SELECT b.id, b.name, b.bullet_diameter_inches, b.bc_g1_published, b.bc_g7_published
FROM bullet b WHERE (bc_g1_published < 0.1 OR bc_g1_published > 0.8)
   OR (bc_g7_published < 0.05 OR bc_g7_published > 0.45);

-- G1/G7 ratio anomalies (0 found)
SELECT b.id, b.name, bc_g1_published/bc_g7_published AS ratio FROM bullet b
WHERE bc_g1_published IS NOT NULL AND bc_g7_published IS NOT NULL
  AND (bc_g7_published > bc_g1_published OR bc_g1_published/bc_g7_published NOT BETWEEN 1.5 AND 2.5);

-- Duplicate bullets (many groups, all legitimate different bullet types)
SELECT m.name, b.weight_grains, ROUND(b.bullet_diameter_inches,3), COUNT(*), GROUP_CONCAT(b.name,' | ')
FROM bullet b JOIN manufacturer m ON b.manufacturer_id = m.id
GROUP BY b.manufacturer_id, b.weight_grains, ROUND(b.bullet_diameter_inches,3) HAVING COUNT(*) > 1;

-- Missing BC rifle bullets excl. CE/Nosler (16 found)
SELECT b.id, b.name, m.name FROM bullet b JOIN manufacturer m ON b.manufacturer_id = m.id
WHERE bc_g1_published IS NULL AND bc_g7_published IS NULL AND bc_g1_estimated IS NULL AND bc_g7_estimated IS NULL
  AND bullet_diameter_inches <= 0.375 AND m.name NOT LIKE '%Cutting Edge%' AND m.name NOT LIKE '%Nosler%'
  AND b.id NOT IN (SELECT DISTINCT bullet_id FROM bullet_bc_source);
```
