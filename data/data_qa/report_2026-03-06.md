## Queries Run

### Row counts
```sql
SELECT COUNT(*) FROM manufacturer; -- 86
SELECT COUNT(*) FROM caliber;      -- 116
SELECT COUNT(*) FROM bullet;       -- 770
SELECT COUNT(*) FROM cartridge;    -- 236
```

### Referential integrity
```sql
SELECT 'orphan_cart_bullet', COUNT(*) FROM cartridge c LEFT JOIN bullet b ON c.bullet_id = b.id
WHERE c.bullet_id IS NOT NULL AND b.id IS NULL;  -- 0
-- (same pattern for all FK checks: all 0)
```

### Cartridge-bullet weight mismatch
```sql
SELECT c.name, c.bullet_weight_grains, b.weight_grains, b.name
FROM cartridge c JOIN bullet b ON c.bullet_id = b.id
WHERE ABS(c.bullet_weight_grains - b.weight_grains) > 0.5;  -- 70 rows
```

### Bullet diameter vs caliber diameter
```sql
SELECT c.name, cal.bullet_diameter_inches, b.bullet_diameter_inches
FROM cartridge c JOIN bullet b ON c.bullet_id = b.id JOIN caliber cal ON c.caliber_id = cal.id
WHERE ABS(b.bullet_diameter_inches - cal.bullet_diameter_inches) > 0.002;  -- 0 rows
```

### Duplicate detection
```sql
SELECT m.name, b1.name, b2.name, b1.bullet_diameter_inches, b1.weight_grains
FROM bullet b1 JOIN bullet b2 ON b1.manufacturer_id = b2.manufacturer_id
  AND b1.bullet_diameter_inches = b2.bullet_diameter_inches
  AND b1.weight_grains = b2.weight_grains AND b1.id < b2.id
JOIN manufacturer m ON b1.manufacturer_id = m.id;  -- 70 pairs
```

### BC range checks
```sql
SELECT b.name, m.name, b.bc_g1_published FROM bullet b JOIN manufacturer m ON b.manufacturer_id = m.id
WHERE b.bc_g1_published IS NOT NULL AND (b.bc_g1_published < 0.050 OR b.bc_g1_published > 0.900);
-- 14 G1 outliers (all explainable)
SELECT b.name, m.name, b.bc_g7_published FROM bullet b JOIN manufacturer m ON b.manufacturer_id = m.id
WHERE b.bc_g7_published IS NOT NULL AND b.bc_g7_published > 0.500;
-- 3 G7 outliers (all .510 cal)
```

### G1/G7 ratio
```sql
SELECT b.name, b.bc_g1_published / b.bc_g7_published AS ratio
FROM bullet b WHERE b.bc_g1_published IS NOT NULL AND b.bc_g7_published IS NOT NULL AND b.bc_g7_published > 0
AND (b.bc_g1_published / b.bc_g7_published < 1.5 OR b.bc_g1_published / b.bc_g7_published > 2.5);
-- 0 rows
```

### Missing BC (rifle bullets, excl CE & Nosler)
```sql
SELECT m.name, COUNT(*) FROM bullet b JOIN manufacturer m ON b.manufacturer_id = m.id
WHERE b.bc_g1_published IS NULL AND b.bc_g1_estimated IS NULL
  AND b.bc_g7_published IS NULL AND b.bc_g7_estimated IS NULL
  AND b.bullet_diameter_inches <= 0.375
  AND m.name NOT IN ('Cutting Edge Bullets', 'Nosler')
GROUP BY m.name;
-- Federal 15, Lapua 5, Lehigh 3, Speer 2, Sierra 2, Berger 1 = 28 total
```

### Velocity anomalies

SELECT c.name, c.muzzle_velocity_fps FROM cartridge c WHERE c.muzzle_velocity_fps = 0;  -- 7 rows
SELECT c.name, c.muzzle_velocity_fps FROM cartridge c
