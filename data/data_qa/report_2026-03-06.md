## Queries Run

```sql
-- Referential integrity (5 queries, all returned 0)
SELECT COUNT(*) FROM bullet b LEFT JOIN manufacturer m ON b.manufacturer_id = m.id WHERE m.id IS NULL;
-- (repeated for cartridge→bullet, cartridge→caliber, cartridge→manufacturer, bcs→bullet)

-- Weight mismatches (68 results)
SELECT c.id, c.name, c.bullet_weight_grains, b.weight_grains, ABS(c.bullet_weight_grains - b.weight_grains) AS diff
FROM cartridge c JOIN bullet b ON c.bullet_id = b.id
WHERE ABS(c.bullet_weight_grains - b.weight_grains) > 1.0;

-- Diameter mismatches (0 results)
SELECT c.id, c.name, b.bullet_diameter_inches, cal.bullet_diameter_inches
FROM cartridge c JOIN bullet b ON c.bullet_id = b.id JOIN caliber cal ON c.caliber_id = cal.id
WHERE ABS(b.bullet_diameter_inches - cal.bullet_diameter_inches) > 0.002;

-- Duplicate bullets (56 groups)
SELECT m.name, b.weight_grains, ROUND(b.bullet_diameter_inches, 3), COUNT(*), GROUP_CONCAT(b.name, ' | ')
FROM bullet b JOIN manufacturer m ON b.manufacturer_id = m.id
GROUP BY b.manufacturer_id, b.weight_grains, ROUND(b.bullet_diameter_inches, 3) HAVING COUNT(*) > 1;

-- BC outside typical range (28 results — all contextually valid)
SELECT b.id, b.name, m.name, b.bullet_diameter_inches, b.bc_g1_published, b.bc_g7_published
FROM bullet b JOIN manufacturer m ON b.manufacturer_id = m.id
WHERE (b.bc_g1_published IS NOT NULL AND (b.bc_g1_published < 0.1 OR b.bc_g1_published > 0.8))
   OR (b.bc_g7_published IS NOT NULL AND (b.bc_g7_published < 0.05 OR b.bc_g7_published > 0.45));

-- G1/G7 swap detection (0 results)
SELECT b.id, b.name, b.bc_g1_published, b.bc_g7_published
FROM bullet b WHERE b.bc_g1_published IS NOT NULL AND b.bc_g7_published IS NOT NULL
  AND b.bc_g7_published > b.bc_g1_published;

-- Zero/implausible velocity (8 results: 7 zeros + 1 at 4400fps)
SELECT c.id, c.name, c.muzzle_velocity_fps, m.name, cal.name
FROM cartridge c JOIN manufacturer m ON c.manufacturer_id = m.id JOIN caliber cal ON c.caliber_id = cal.id
WHERE c.muzzle_velocity_fps <= 0 OR c.muzzle_velocity_fps > 4200;

-- Missing BC rifle bullets (28 results)
SELECT b.id, b.name, m.name, b.bullet_diameter_inches
FROM bullet b JOIN manufacturer m ON b.manufacturer_id = m.id
WHERE b.bc_g1_published IS NULL AND b.bc_g7_published IS NULL
  AND b.bc_g1_estimated IS NULL AND b.bc_g7_estimated IS NULL
  AND b.bullet_diameter_inches <= 0.375 AND m.name NOT IN ('Cutting Edge Bullets', 'Nosler');

-- Unit confusion (1 result: 30-06 220gr Custom International at MV=721, barrel=61)
SELECT c.id, c.name, c.muzzle_velocity_fps, c.test_barrel_length_inches
FROM cartridge c WHERE c.test_barrel_length_inches > 40
   OR (c.muzzle_velocity_fps BETWEEN 1 AND 1000 AND c.name NOT LIKE '%Blackout%' AND c.name NOT LIKE '%Sub%');
```
