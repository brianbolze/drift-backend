## Queries Run

```sql
-- Referential integrity (all returned 0)
SELECT COUNT(*) FROM bullet b LEFT JOIN manufacturer m ON b.manufacturer_id = m.id WHERE m.id IS NULL;
SELECT COUNT(*) FROM cartridge c LEFT JOIN bullet b ON c.bullet_id = b.id WHERE c.bullet_id IS NOT NULL AND b.id IS NULL;
SELECT COUNT(*) FROM cartridge c LEFT JOIN caliber cal ON c.caliber_id = cal.id WHERE cal.id IS NULL;
SELECT COUNT(*) FROM cartridge c LEFT JOIN manufacturer m ON c.manufacturer_id = m.id WHERE m.id IS NULL;
SELECT COUNT(*) FROM bullet_bc_source bcs LEFT JOIN bullet b ON bcs.bullet_id = b.id WHERE b.id IS NULL;

-- Weight mismatches (returned 100)
SELECT COUNT(*) FROM cartridge c JOIN bullet b ON c.bullet_id = b.id
WHERE ABS(c.bullet_weight_grains - b.weight_grains) > 1.0;

-- Diameter mismatches (returned 0)
SELECT COUNT(*) FROM cartridge c JOIN bullet b ON c.bullet_id = b.id JOIN caliber cal ON c.caliber_id = cal.id
WHERE ABS(b.bullet_diameter_inches - cal.bullet_diameter_inches) > 0.002;

-- BC outside hard error thresholds (returned 2)
SELECT b.id, b.name, b.bc_g1_published, b.bc_g7_published FROM bullet b
WHERE (b.bc_g1_published < 0.05 OR b.bc_g1_published > 1.2) OR (b.bc_g7_published < 0.02 OR b.bc_g7_published > 0.6);

-- G1/G7 swaps (returned 0)
SELECT b.id FROM bullet b WHERE b.bc_g1_published IS NOT NULL AND b.bc_g7_published IS NOT NULL
AND b.bc_g7_published > b.bc_g1_published;

-- BC ratio outliers (returned 0)
SELECT b.id FROM bullet b WHERE b.bc_g1_published IS NOT NULL AND b.bc_g7_published IS NOT NULL
AND (b.bc_g1_published / b.bc_g7_published < 1.5 OR b.bc_g1_published / b.bc_g7_published > 2.5);

-- Zero/implausible velocity (returned 8)
SELECT c.id, c.name, c.muzzle_velocity_fps FROM cartridge c
WHERE c.muzzle_velocity_fps <= 0 OR c.muzzle_velocity_fps > 4200;

-- Missing BC rifle bullets (returned 6)
SELECT b.id, b.name, m.name FROM bullet b JOIN manufacturer m ON b.manufacturer_id = m.id
WHERE b.bc_g1_published IS NULL AND b.bc_g7_published IS NULL AND b.bc_g1_estimated IS NULL AND b.bc_g7_estimated IS NULL
AND b.id NOT IN (SELECT DISTINCT bullet_id FROM bullet_bc_source)
AND m.name NOT LIKE '%Cutting Edge%' AND m.name NOT LIKE '%Nosler%' AND b.bullet_diameter_inches <= 0.375;
```
